Question: Write a C++ class called BankAccount that demonstrates the use of default constructor, parameterized constructor, copy constructor, and destructor. The class should have member variables for account number, account holder name, and balance. Explain each type of constructor with an example.

Answer (English):

#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    int accountNumber;
    string holderName;
    double balance;

public:
    // Default Constructor
    BankAccount() {
        accountNumber = 0;
        holderName = "Unknown";
        balance = 0.0;
        cout << "Default constructor called" << endl;
    }

    // Parameterized Constructor
    BankAccount(int accNum, string name, double bal) {
        accountNumber = accNum;
        holderName = name;
        balance = bal;
        cout << "Parameterized constructor called" << endl;
    }

    // Copy Constructor
    BankAccount(const BankAccount &acc) {
        accountNumber = acc.accountNumber;
        holderName = acc.holderName;
        balance = acc.balance;
        cout << "Copy constructor called" << endl;
    }

    // Destructor
    ~BankAccount() {
        cout << "Destructor called for account: "
             << accountNumber << endl;
    }

    void display() {
        cout << "Account: " << accountNumber
             << ", Holder: " << holderName
             << ", Balance: $" << balance << endl;
    }
};

// Example Usage
int main() {
    BankAccount acc1;  // Default constructor
    BankAccount acc2(12345, "John Doe", 5000.0);  // Parameterized
    BankAccount acc3 = acc2;  // Copy constructor
    
    acc2.display();
    return 0;
}
Explanation:
  • Default Constructor: Called when an object is created without parameters. Initializes member variables with default values.
  • Parameterized Constructor: Accepts arguments to initialize object with specific values at creation time.
  • Copy Constructor: Creates a new object as a copy of an existing object. Takes a constant reference to another object of the same class.
  • Destructor: Automatically called when an object goes out of scope or is deleted. Used for cleanup operations like releasing memory.

class Person {
protected:
    string name;
    int age;
    string id;

public:
    Person() : name(""), age(0), id("") {
        cout << "Person default constructor" << endl;
    }
    
    Person(string n, int a, string i)
        : name(n), age(a), id(i) {
        cout << "Person parameterized constructor" << endl;
    }

    void setName(string n) {
        name = n;
    }
    
    string getName() {
        return name;
    }

    virtual void display() {
        cout << "Name: " << name
             << ", Age: " << age
             << ", ID: " << id << endl;
    }
};

// Derived Class - Student
class Student : public Person {
private:
    string grade;
    double gpa;

public:
    Student() : Person(), grade(""), gpa(0.0) {
        cout << "Student default constructor" << endl;
    }
    
    Student(string n, int a, string i,
            string g, double gp)
        : Person(n, a, i), grade(g), gpa(gp) {
        cout << "Student parameterized constructor" << endl;
    }

    void study() {
        cout << name << " is studying." << endl;
    }

    void display() override {
        Person::display();
        cout << "Grade: " << grade
             << ", GPA: " << gpa << endl;
    }
};

// Derived Class - Teacher
class Teacher : public Person {
private:
    string subject;
    double salary;

public:
    Teacher() : Person(), subject(""), salary(0.0) {
        cout << "Teacher default constructor" << endl;
    }
    
    Teacher(string n, int a, string i,
             string s, double sal)
        : Person(n, a, i), subject(s), salary(sal) {
        cout << "Teacher parameterized constructor" << endl;
    }

    void teach() {
        cout << name << " is teaching "
             << subject << endl;
    }

    void display() override {
        Person::display();
        cout << "Subject: " << subject
             << ", Salary: $" << salary << endl;
    }
};

// Example Usage
int main() {
    Student s1("Alice", 20, "S001", "A", 3.8);
    Teacher t1("Dr. Smith", 45, "T001", "Math", 50000);
    
    s1.display();
    s1.study();

}

  • Protected Members: Base class members accessible to derived classes
  • Virtual Functions: display() method can be overridden in derived classes
  • Constructor Chaining: Derived class constructors call base class constructor
  • Code Reusability: Common attributes (name, age, id) defined once in base class

Question: Explain the concept of encapsulation in C++ with a practical example. Create a class Employee that demonstrates data hiding using private members and public getter/setter methods. Discuss the importance of encapsulation in software design.
English

Answer (English):

#include <iostream>
#include <string>
using namespace std;

class Employee {
private:
    // Private data members (Hidden from outside)
    string name;
    int empID;
    double salary;
    string department;

    // Private helper method
    bool isValidSalary(double sal) {
        return sal >= 0 && sal <= 1000000;
    }

public:
    // Constructor
    Employee(string n, int id, double sal, string dept) {
        name = n;
        empID = id;
        setSalary(sal);  // Using setter for validation
        department = dept;
    }

    // Getter methods (Read access)
    string getName() const {
        return name;
    }

    int getEmpID() const {
        return empID;
    }

    double getSalary() const {
        return salary;
    }

    string getDepartment() const {
        return department;
    }

    // Setter methods (Write access with validation)
    void setName(string n) {
        if (!n.empty()) {
            name = n;
        } else {
            cout << "Invalid name!" << endl;
        }
    }

    void setSalary(double sal) {
        if (isValidSalary(sal)) {
            salary = sal;
        } else {
            cout << "Invalid salary! Setting to 0." << endl;
            salary = 0;
        }
    }

    void setDepartment(string dept) {
        department = dept;
    }

    // Method to give raise
    void giveRaise(double percentage) {
        double increment = salary * (percentage / 100);
        setSalary(salary + increment);
        cout << "Salary increased by " << percentage
             << "%" << endl;
    }

    // Display method
    void displayInfo() const {
        cout << "Employee ID: " << empID << endl;
        cout << "Name: " << name << endl;
        cout << "Department: " << department << endl;
        cout << "Salary: $" << salary << endl;
    }
};

// Example Usage
int main() {
    Employee emp1("John Doe", 101, 50000, "IT");
    
    emp1.displayInfo();
    emp1.giveRaise(10);
    emp1.displayInfo();
    
    // Cannot access: emp1.salary = 100000; // ERROR!
    // Must use: emp1.setSalary(100000); // Correct!
    
    return 0;
}
Importance of Encapsulation:
  • Data Protection: Private members cannot be accessed directly, preventing unauthorized modifications
  • Validation: Setter methods can validate data before storing, ensuring data integrity
  • Flexibility: Internal implementation can change without affecting external code
  • Maintainability: Changes to private members don't break existing code using the class
  • Security: Sensitive information like salary remains hidden and controlled
  • Modularity: Creates clear interfaces between different parts of the program
Access Specifiers in C++:
  • private: Members accessible only within the class
  • public: Members accessible from anywhere
  • protected: Members accessible within class and derived classes

#include <iostream>
#include <cmath>
using namespace std;

// Base Class - Demonstrating Runtime Polymorphism
class Shape {
protected:
    string color;

public:
    Shape(string c = "White") : color(c) {}

    // Pure virtual function (makes class abstract)
    virtual double calculateArea() = 0;
    
    virtual void display() {
        cout << "Shape Color: " << color << endl;
    }

    virtual ~Shape() {
        cout << "Shape destructor" << endl;
    }
};

// Derived Class - Circle
class Circle : public Shape {
private:
    double radius;

public:
    Circle(double r, string c = "Red")
        : Shape(c), radius(r) {}

    // Overriding virtual function
    double calculateArea() override {
        return 3.14159 * radius * radius;
    }

    void display() override {
        Shape::display();
        cout << "Circle - Radius: " << radius
             << ", Area: " << calculateArea() << endl;
    }

    // Function Overloading - Compile-time Polymorphism
    void resize(double factor) {
        radius *= factor;
        cout << "Circle resized by factor: "
             << factor << endl;
    }

    void resize(double newRadius, bool absolute) {
        if (absolute) {
            radius = newRadius;
            cout << "Circle radius set to: "
                 << newRadius << endl;
        }
    }
};

// Derived Class - Rectangle
class Rectangle : public Shape {
private:
    double length;
    double width;

public:
    Rectangle(double l, double w, string c = "Blue")
        : Shape(c), length(l), width(w) {}

    // Overriding virtual function
    double calculateArea() override {
        return length * width;
    }

    void display() override {
        Shape::display();
        cout << "Rectangle - Length: " << length
             << ", Width: " << width
             << ", Area: " << calculateArea() << endl;
    }

    // Operator Overloading - Compile-time Polymorphism
    Rectangle operator+(const Rectangle& r) {
        return Rectangle(length + r.length,
                          width + r.width, color);
    }

    bool operator==(const Rectangle& r) {
        return (length == r.length && width == r.width);
    }
};

// Demonstration of Function Overloading
class Calculator {
public:
    int add(int a, int b) {
        return a + b;
    }

    double add(double a, double b) {
        return a + b;
    }

    int add(int a, int b, int c) {
        return a + b + c;
    }
};

// Main function demonstrating polymorphism
int main() {
    // Runtime Polymorphism
    cout << "=== Runtime Polymorphism ===" << endl;
    Shape* shapes[2];
    shapes[0] = new Circle(5.0, "Red");
    shapes[1] = new Rectangle(4.0, 6.0, "Blue");

    for(int i = 0; i < 2; i++) {
        shapes[i]->display();
        cout << endl;
    }

    // Compile-time Polymorphism - Function Overloading
    cout << "=== Function Overloading ===" << endl;
    Calculator calc;
    cout << "Add(2, 3): " << calc.add(2, 3) << endl;
    cout << "Add(2.5, 3.7): " << calc.add(2.5, 3.7) << endl;
    cout << "Add(1, 2, 3): " << calc.add(1, 2, 3) << endl;

    // Cleanup
    for(int i = 0; i < 2; i++) {
        delete shapes[i];
    }

    return 0;
}
Types of Polymorphism:

1. Compile-time Polymorphism (Static Binding):

  • Function Overloading: Multiple functions with same name but different parameters
  • Operator Overloading: Redefining operators (+, ==, etc.) for custom classes
  • Resolved at compile time
  • Faster execution as no runtime overhead

2. Runtime Polymorphism (Dynamic Binding):

  • Virtual Functions: Functions declared with 'virtual' keyword in base class
  • Function Overriding: Derived class provides specific implementation
  • Resolved at runtime using vtable (virtual table)
  • Enables dynamic method dispatch
  • Base class pointer can call derived class methods
Key Benefits:
  • Code reusability and flexibility
  • Loose coupling between objects
  • Easier to extend and maintain code
  • Implements "one interface, multiple implementations" principle

#include <iostream>
#include <string>
using namespace std;

// Abstract Base Class
class Vehicle {
protected:
    string brand;
    string model;
    int year;

public:
    // Constructor
    Vehicle(string b, string m, int y)
        : brand(b), model(m), year(y) {
        cout << "Vehicle constructor called" << endl;
    }

    // Pure virtual functions (makes class abstract)
    virtual void startEngine() = 0;
    virtual void stopEngine() = 0;
    virtual void displayInfo() = 0;

    // Concrete method
    string getBrand() const {
        return brand;
    }

    // Virtual destructor
    virtual ~Vehicle() {
        cout << "Vehicle destructor called" << endl;
    }
};

// Concrete Derived Class - Car
class Car : public Vehicle {
private:
    int numberOfDoors;
    bool hasTrunk;

public:
    Car(string b, string m, int y,
        int doors, bool trunk)
        : Vehicle(b, m, y),
          numberOfDoors(doors),
          hasTrunk(trunk) {
        cout << "Car constructor called" << endl;
    }

    // Implementing pure virtual functions
    void startEngine() override {
        cout << "Car engine started with key ignition"
             << endl;
        cout << "Vroom vroom! üöó" << endl;
    }

    void stopEngine() override {
        cout << "Car engine stopped" << endl;
    }

    void displayInfo() override {
        cout << "\n=== Car Information ===" << endl;
        cout << "Brand: " << brand << endl;
        cout << "Model: " << model << endl;
        cout << "Year: " << year << endl;
        cout << "Doors: " << numberOfDoors << endl;
        cout << "Has Trunk: "
             << (hasTrunk ? "Yes" : "No") << endl;
    }

    // Car-specific method
    void openTrunk() {
        if (hasTrunk) {
            cout << "Trunk opened" << endl;
        } else {
            cout << "This car has no trunk" << endl;
        }
    }
};

// Concrete Derived Class - Motorcycle

    int engineCC;

    Motorcycle(string b, string m, int y,

int main() {
    // Cannot create object of abstract class

Concept of Abstraction:

Abstraction is the process of hiding implementation details and showing only essential features of an object. It focuses on WHAT an object does rather than HOW it does it.

Key Points about Abstract Classes:

  • Pure Virtual Function: Declared by assigning = 0, has no implementation in base class
  • Abstract Class: A class with at least one pure virtual function
  • Cannot Instantiate: Objects of abstract classes cannot be created directly
  • Must Override: Derived classes must implement all pure virtual functions
  • Interface Definition: Abstract classes define a contract that derived classes must follow

Benefits of Abstraction:

  • Reduces Complexity: Users only see relevant features
  • Increases Security: Implementation details are hidden
  • Code Reusability: Common interface for different implementations
  • Flexibility: Easy to add new implementations without changing existing code
  • Maintainability: Changes in implementation don't affect users of the class

Real-world Analogy:

Think of a TV remote control - you know WHAT buttons to press to change channels, but you don't need to know HOW the signal is transmitted or processed internally. That's abstraction!
