1. Write a program to find all possible pairs of elements in an array that sum up to a given target value ‚ñº 
// Find Pairs with Given Sum (Two-Sum)
#include <iostream>
using namespace std;

void findPairs(int arr[], int n, int target) {
    cout << "Pairs that sum to " << target << ":" << endl;
    bool found = false;
    
    // Use nested loops to check every possible pair (O(n^2))
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            if (arr[i] + arr[j] == target) {
                cout << "(" << arr[i] << ", " << arr[j] << ") at indices (" << i << ", " << j << ")" << endl;
                found = true;
            }
        }
    }
    
    if (!found) {
        cout << "No pairs found." << endl;
    }
}

int main() {
    int arr[] = {1, 5, 7, -1, 5};
    int n = sizeof(arr) / sizeof(arr[0]);
    int sum = 6;
    
    findPairs(arr, n, sum);
    
    return 0;
}

Explanation:

This program implements the brute-force approach for the Two-Sum problem. It uses nested loops to check every unique pair of elements (arr[i] and arr[j] where j > i). If the sum of the pair equals the target value, the pair is printed.

3. Implement a function to find the $N^{th}$ power of a number using recursion ‚ñº 
// Power Calculation using Recursion
#include <iostream>
using namespace std;

// Recursive function to calculate base^exp
long long power(int base, int exp) {
    // Base Case 1: Any number to the power of 0 is 1
    if (exp == 0) {
        return 1;
    }
    // Base Case 2: Any number to the power of 1 is the base itself
    if (exp == 1) {
        return base;
    }
    // Recursive Step: base^exp = base * base^(exp-1)
    return base * power(base, exp - 1);
}

int main() {
    int b1 = 2, e1 = 10;
    int b2 = 5, e2 = 3;

    cout << b1 << " raised to " << e1 << " is: " << power(b1, e1) << endl; // 2^10 = 1024
    cout << b2 << " raised to " << e2 << " is: " << power(b2, e2) << endl; // 5^3 = 125
    
    return 0;
}

Explanation:

This function uses recursion based on the mathematical identity $x^n = x \times x^{n-1}$. The recursive calls break down the problem until the base case $x^0 = 1$ or $x^1 = x$ is reached. The function then returns the result of the cumulative multiplications.

// Perfect Number Check
#include <iostream>
using namespace std;

bool isPerfect(int n) {
    if (n <= 1) return false;
    
    int sumOfDivisors = 0;
    
    // Check for divisors from 1 up to n-1
    for (int i = 1; i < n; i++) {
        if (n % i == 0) {

        }
    }
    
    // A number is perfect if the sum of its proper positive divisors equals the number itself

}

    return 0;
}

Explanation:

A Perfect Number is a positive integer that is equal to the sum of its proper positive divisors (divisors excluding the number itself). The function iterates from 1 up to $n-1$, checks for divisibility, sums the divisors, and finally compares the sum to the original number.

7. Implement a basic Queue data structure using an array (FIFO principle) ‚ñº 
// Queue Implementation (Array-based)
#include <iostream>
using namespace std;

const int MAX_SIZE = 5;

class Queue {
private:
    int arr[MAX_SIZE];
    int front;
    int rear;

public:
    // Constructor: Initialize front and rear to -1 (empty queue)
    Queue() : front(-1), rear(-1) {} 
    
    bool isEmpty() const {
        return front == -1;
    }
    
    bool isFull() const {
        return rear == MAX_SIZE - 1;
    }
    
    // Enqueue operation (add to rear)
    void enqueue(int value) {
        if (isFull()) {
            cout << "Queue Overflow: Cannot enqueue " << value << endl;
            return;
        }
        if (isEmpty()) {
            front = 0; // Set front to 0 for first element
        }
        arr[++rear] = value;
        cout << "Enqueued: " << value << endl;
    }
    
    // Dequeue operation (remove from front)
    int dequeue() {
        if (isEmpty()) {
            cout << "Queue Underflow" << endl;
            return -1; // Error code
        }
        int value = arr[front++];
        
        // Reset queue if the last element was dequeued
        if (front > rear) {
            front = rear = -1;
        }
        return value;
    }
};

int main() {
    Queue q;
    
    q.enqueue(100);
    q.enqueue(200);
    
    cout << "Dequeued: " << q.dequeue() << endl; // 100
    
    q.enqueue(300);
    q.enqueue(400);
    q.enqueue(500);
    q.enqueue(600); // Should cause overflow
    
    cout << "Dequeued: " << q.dequeue() << endl; // 200
    
    return 0;
}

Explanation:

The queue is a First In, First Out (FIFO) data structure. This array-based implementation uses two pointers, front and rear. enqueue adds to the rear, and dequeue removes from the front. The pointers are managed carefully to handle the empty and full states. (Note: A Circular Queue is a more efficient array-based solution, but this linear version demonstrates the basic principle.)

string mergeAlternatively(const string& s1, const string& s2) {
    string result = "";
    int len1 = s1.length();
    int len2 = s2.length();
    int maxLen = max(len1, len2);

    for (int i = 0; i < maxLen; i++) {
        // Append from s1 if index is valid
        if (i < len1) {
            result += s1[i];
        }
        // Append from s2 if index is valid

The function determines the length of the longer string. It then iterates up to that maximum length. In each iteration, it conditionally appends one character from each string if the index is within the respective string's bounds. This handles strings of unequal lengths correctly.

12. Implement a function to check if two strings are Anagrams (e.g., "listen" and "silent") ‚ñº 
// Anagram Check using Character Counts
#include <iostream>
include <string>
include <algorithm>
include <cctype>
using namespace std;

// Assume a small alphabet size for simplicity (e.g., 26)
const int CHAR_RANGE = 256; 

bool areAnagrams(const string& s1, const string& s2) {
    // Trivial case: lengths must be equal
    if (s1.length() != s2.length()) {
        return false;
    }

    // Frequency array for 256 ASCII characters
    int count[CHAR_RANGE] = {0};

    // Increment count for characters in s1, decrement for characters in s2
    for (int i = 0; i < s1.length(); i++) {
        count[(int)s1[i]]++;
        count[(int)s2[i]]--;
    }

    // If strings are anagrams, all counts must be 0
    for (int i = 0; i < CHAR_RANGE; i++) {
        if (count[i] != 0) {
            return false;
        }
    }

    return true;
}

int main() {
    string strA = "listen";
    string strB = "silent";
    string strC = "hello";
    
    cout << "\"" << strA << "\" and \"" << strB << "\" are anagrams: " << (areAnagrams(strA, strB) ? "Yes" : "No") << endl;
    cout << "\"" << strA << "\" and \"" << strC << "\" are anagrams: " << (areAnagrams(strA, strC) ? "Yes" : "No") << endl;
    
    return 0;
}

Explanation:

Two strings are anagrams if they contain the exact same characters with the same frequency. This efficient solution uses a fixed-size array (256 for all ASCII chars) as a frequency map. It increments the count for characters in the first string and decrements for the second. If the two strings are anagrams, all counts in the array must be zero at the end.

13. Calculate the GCD of two numbers using an Iterative approach (Euclidean Algorithm) ‚ñº 
// GCD using Iteration (Euclidean Algorithm)
#include <iostream>
using namespace std;

int gcdIterative(int a, int b) {
    while (b != 0) {
        // Calculate remainder (a % b)
        int temp = b;
        // New b is the old remainder (a % b)
        b = a % b; 
        // New a is the old b
        a = temp;
    }
    // When b becomes 0, a is the GCD
    return a;
}

int main() {
    int num1 = 48;
    int num2 = 18;
    
    cout << "The GCD of " << num1 << " and " << num2 << " is: " << gcdIterative(num1, num2) << endl; // Expected: 6
    
    num1 = 60;
    num2 = 48;
    
    cout << "The GCD of " << num1 << " and " << num2 << " is: " << gcdIterative(num1, num2) << endl; // Expected: 12

    return 0;
}

Explanation:

The iterative Euclidean algorithm is highly efficient for finding the Greatest Common Divisor. It repeatedly replaces the larger number with the remainder of the division between the two numbers, until one of the numbers becomes 0. The non-zero number at that point is the GCD.

16. Create a class Counter and overload the pre-increment operator (++) as a member function ‚ñº 
// Unary Operator Overloading (Pre-increment)
#include <iostream>
using namespace std;

class Counter {
private:
    int count;

public:
    Counter(int c = 0) : count(c) {}
    
    // Overload the pre-increment operator (Prefix form: ++obj)
    Counter& operator++() {
        // 1. Increment the value of the object
        ++count;
        // 2. Return a reference to the modified object
        return *this;
    }

    void displayCount() const {
        cout << "Current Count: " << count << endl;
    }
};

int main() {
    Counter c1(10);
    
    cout << "Initial "; c1.displayCount();
    
    // Use the overloaded pre-increment operator
    Counter c2 = ++c1; // c1 is incremented, then assigned to c2
    
    cout << "After ++c1:" << endl;
    cout << "c1 "; c1.displayCount(); // 11
    cout << "c2 "; c2.displayCount(); // 11
    
    // Chaining also works because it returns a reference
    ++(++c1);
    cout << "After ++(++c1): "; c1.displayCount(); // 13
    
    return 0;
}

Explanation:

The pre-increment operator (`++obj`) is overloaded as a member function with no arguments. Crucially, it returns a reference (`Counter&`) to the object itself (`*this`) after incrementing the internal state. This behavior mimics built-in types and allows for chaining of increment operations.

    // Check for invalid position
    if (position < 0 || position >= size) {
        cout << "Invalid position." << endl;
        return size;

// Function to check for leap year condition
bool isLeapYear(int year) {
    // Leap Year Rule:
    // 1. A year is a leap year if it is divisible by 4.
    // 2. EXCEPT if it is divisible by 100.
    // 3. UNLESS it is divisible by 400.
    
    return ((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0);
}

int main() {
    int year1 = 2000; // Divisible by 400 (Leap)
    int year2 = 1900; // Divisible by 100 but not 400 (Not Leap)
    int year3 = 2024; // Divisible by 4 but not 100 (Leap)
    int year4 = 2023; // Not divisible by 4 (Not Leap)
    
    cout << year1 << " is a leap year: " << (isLeapYear(year1) ? "Yes" : "No") << endl;
    cout << year2 << " is a leap year: " << (isLeapYear(year2) ? "Yes" : "No") << endl;
    cout << year3 << " is a leap year: " << (isLeapYear(year3) ? "Yes" : "No") << endl;
    cout << year4 << " is a leap year: " << (isLeapYear(year4) ? "Yes" : "No") << endl;
    
    return 0;
}

Explanation:

The leap year rule is implemented using a concise boolean expression. It captures the three conditions: must be divisible by 4 AND not divisible by 100, OR it must be divisible by 400. This logic accurately determines if a given year is a leap year.
