1. Write a program to find the factorial of a number using recursion â–¼ 
// Program to calculate factorial using recursion
#include <iostream>
using namespace std;

// Recursive function to calculate factorial
int factorial(int n) {
    // Base case: factorial of 0 or 1 is 1
    if (n == 0 || n == 1) {
        return 1;
    }
    // Recursive case: n! = n * (n-1)!
    return n * factorial(n - 1);
}

int main() {
    int num;
    
    // Input from user
    cout << "Enter a positive integer: ";
    cin >> num;
    
    // Validation check
    if (num < 0) {
        cout << "Factorial is not defined for negative numbers!" << endl;
    } else {
        // Calculate and display result
        cout << "Factorial of " << num << " = " << factorial(num) << endl;
    }
    
    return 0;
}

Explanation:

This program uses recursion to calculate factorial. The base case returns 1 when n is 0 or 1. For other values, it multiplies n with factorial of (n-1).

3. Write a program to implement bubble sort algorithm â–¼ 
// Bubble Sort Implementation
#include <iostream>
using namespace std;

// Function to perform bubble sort
void bubbleSort(int arr[], int n) {
    // Outer loop for number of passes
    for (int i = 0; i < n - 1; i++) {
        // Inner loop for comparisons in each pass
        for (int j = 0; j < n - i - 1; j++) {
            // Swap if current element is greater than next
            if (arr[j] > arr[j + 1]) {
                // Swap elements
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

// Function to display array
void displayArray(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    // Display original array
    cout << "Original array: ";
    displayArray(arr, n);
    
    // Sort the array
    bubbleSort(arr, n);
    
    // Display sorted array
    cout << "Sorted array: ";
    displayArray(arr, n);
    
    return 0;
}

Explanation:

Bubble sort repeatedly compares adjacent elements and swaps them if they're in wrong order. Time complexity: O(nÂ²).

// Singly Linked List Implementation
#include <iostream>
using namespace std;

// Node structure
struct Node {
    int data;
    Node* next;
};

// LinkedList class
class LinkedList {
private:
    Node* head;

public:
    // Constructor
    LinkedList() {
        head = nullptr;
    }
    
    // Insert at beginning
    void insertAtBeginning(int value) {
        // Create new node
        Node* newNode = new Node();
        newNode->data = value;
        newNode->next = head;
        
        // Update head
        head = newNode;
        
        cout << "Inserted " << value << " at beginning" << endl;
    }
    
    // Insert at end
    void insertAtEnd(int value) {
        // Create new node
        Node* newNode = new Node();
        newNode->data = value;
        newNode->next = nullptr;
        
        // If list is empty
        if (head == nullptr) {
            head = newNode;
            return;
        }
        
        // Traverse to last node
        Node* temp = head;
        while (temp->next != nullptr) {
            temp = temp->next;
        }
        
        // Insert at end
        temp->next = newNode;
        
        cout << "Inserted " << value << " at end" << endl;
    }
    
    // Display linked list
    void display() {
        if (head == nullptr) {
            cout << "List is empty!" << endl;
            return;
        }
        
        Node* temp = head;
        cout << "Linked List: ";
        
        while (temp != nullptr) {
            cout << temp->data << " -> ";
            temp = temp->next;
        }
        
        cout << "NULL" << endl;
    }
};

int main() {
    LinkedList list;
    
    // Insert elements
    list.insertAtEnd(10);
    list.insertAtEnd(20);
    list.insertAtBeginning(5);
    list.insertAtEnd(30);
    
    // Display list
    list.display();
    
    return 0;
}

7. Write a program to implement binary search algorithm â–¼ 
// Binary Search Implementation
#include <iostream>
using namespace std;

// Iterative binary search function
int binarySearch(int arr[], int n, int target) {
    int left = 0;
    int right = n - 1;
    
    // Continue searching while search space exists
    while (left <= right) {
        // Calculate middle index
        int mid = left + (right - left) / 2;
        
        // Check if target is at middle
        if (arr[mid] == target) {
            return mid;
        }
        
        // If target is greater, ignore left half
        if (arr[mid] < target) {
            left = mid + 1;
        }
        // If target is smaller, ignore right half
        else {
            right = mid - 1;
        }
    }
    
    // Element not found
    return -1;
}

int main() {
    int arr[] = {2, 5, 8, 12, 16, 23, 38, 45, 56, 67, 78};
    int n = sizeof(arr) / sizeof(arr[0]);
    int target = 23;
    
    // Perform binary search
    int result = binarySearch(arr, n, target);
    
    if (result != -1) {
        cout << "Element found at index: " << result << endl;
    } else {
        cout << "Element not found in array" << endl;
    }
    
    return 0;
}

Explanation:

Binary search works on sorted arrays by repeatedly dividing search space in half. Time complexity: O(log n).

9. Write a program to implement a stack using arrays â–¼ 
// Stack Implementation Using Array
#include <iostream>
using namespace std;

#define MAX 100

class Stack {
private:
    int arr[MAX];
    int top;

public:
    // Constructor
    Stack() {
        top = -1;
    }
    
    // Check if stack is empty
    bool isEmpty() {
        return (top == -1);
    }
    
    // Check if stack is full
    bool isFull() {
        return (top == MAX - 1);
    }
    
    // Push element onto stack
    void push(int value) {
        if (isFull()) {
            cout << "Stack Overflow! Cannot push " << value << endl;
            return;
        }
        
        // Increment top and add element
        top++;
        arr[top] = value;
        cout << "Pushed " << value << " onto stack" << endl;
    }
    
    // Pop element from stack
    int pop() {
        if (isEmpty()) {
            cout << "Stack Underflow! Cannot pop" << endl;
            return -1;
        }
        
        // Get top element and decrement top
        int value = arr[top];
        top--;
        cout << "Popped " << value << " from stack" << endl;
        return value;
    }
    
    // Peek at top element
    int peek() {
        if (isEmpty()) {
            cout << "Stack is empty!" << endl;
            return -1;
        }
        return arr[top];
    }
    
    // Display stack contents
    void display() {
        if (isEmpty()) {
            cout << "Stack is empty!" << endl;
            return;
        }
        
        cout << "Stack contents (top to bottom): ";
        for (int i = top; i >= 0; i--) {
            cout << arr[i] << " ";
        }
        cout << endl;
    }
};

int main() {
    Stack s;
    
    // Perform stack operations
    s.push(10);
    s.push(20);
    s.push(30);
    s.push(40);
    
    s.display();
    
    cout << "Top element: " << s.peek() << endl;
    
    s.pop();
    s.pop();
    
    s.display();
    
    return 0;
}

Explanation:

Stack follows LIFO (Last In First Out) principle. Implements push, pop, peek, and display operations using array.

11. Write a program to implement a queue using arrays â–¼ 
// Queue Implementation Using Array
#include <iostream>
using namespace std;

#define MAX 100

class Queue {
private:
    int arr[MAX];
    int front;
    int rear;
    int count;

public:
    // Constructor
    Queue() {
        front = 0;
        rear = -1;
        count = 0;
    }
    
    // Check if queue is empty
    bool isEmpty() {
        return (count == 0);
    }
    
    // Check if queue is full
    bool isFull() {
        return (count == MAX);
    }
    
    // Enqueue operation
    void enqueue(int value) {
        if (isFull()) {
            cout << "Queue Overflow! Cannot enqueue " << value << endl;
            return;
        }
        
        // Circular increment of rear
        rear = (rear + 1) % MAX;
        arr[rear] = value;
        count++;
        
        cout << "Enqueued " << value << endl;
    }
    
    // Dequeue operation
    int dequeue() {
        if (isEmpty()) {
            cout << "Queue Underflow! Cannot dequeue" << endl;
            return -1;
        }
        
        // Get front element
        int value = arr[front];
        
        // Circular increment of front
        front = (front + 1) % MAX;
        count--;
        
        cout << "Dequeued " << value << endl;
        return value;
    }
    
    // Peek at front element
    int peek() {
        if (isEmpty()) {
            cout << "Queue is empty!" << endl;
            return -1;
        }
        return arr[front];
    }
    
    // Display queue contents
    void display() {
        if (isEmpty()) {
            cout << "Queue is empty!" << endl;
            return;
        }
        
        cout << "Queue contents (front to rear): ";
        int i = front;
        for (int j = 0; j < count; j++) {
            cout << arr[i] << " ";
            i = (i + 1) % MAX;
        }
        cout << endl;
    }
};

int main() {
    Queue q;
    
    // Perform queue operations
    q.enqueue(10);
    q.enqueue(20);
    q.enqueue(30);
    q.enqueue(40);
    
    q.display();
    
    cout << "Front element: " << q.peek() << endl;
    
    q.dequeue();
    q.dequeue();
    
    q.display();
    
    q.enqueue(50);
    q.display();
    
    return 0;
}

Explanation:

Queue follows FIFO (First In First Out) principle. Implements enqueue, dequeue using circular array for efficient space utilization.

// Matrix Multiplication (3x3)
#include <iostream>
using namespace std;

const int SIZE = 3;

// Function to multiply two matrices
void multiplyMatrices(int mat1[SIZE][SIZE], int mat2[SIZE][SIZE], int result[SIZE][SIZE]) {
    // Initialize result matrix to 0
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            result[i][j] = 0;
        }
    }

    // Perform multiplication: result[i][j] = sum(mat1[i][k] * mat2[k][j])
    for (int i = 0; i < SIZE; i++) { // Rows of mat1
        for (int j = 0; j < SIZE; j++) { // Columns of mat2
            for (int k = 0; k < SIZE; k++) { // Inner dimension
                result[i][j] += mat1[i][k] * mat2[k][j];
            }
        }
    }
}

// Function to display a matrix
void displayMatrix(int mat[SIZE][SIZE]) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            cout << mat[i][j] << "\t";
        }
        cout << endl;
    }
}

int main() {
    int matA[SIZE][SIZE] = { {1, 2, 3}, {4, 5, 6}, {7, 8, 9} };
    int matB[SIZE][SIZE] = { {9, 8, 7}, {6, 5, 4}, {3, 2, 1} };
    int resultMat[SIZE][SIZE];
    
    cout << "Matrix A:" << endl;
    displayMatrix(matA);
    
    cout << "\nMatrix B:" << endl;
    displayMatrix(matB);
    
    multiplyMatrices(matA, matB, resultMat);
    
    cout << "\nResult of A * B:" << endl;
    displayMatrix(resultMat);
    
    return 0;
}

Explanation:

Matrix multiplication involves a nested loop structure. The outermost loops iterate over the rows of the first matrix (i) and columns of the second matrix (j). The innermost loop (k) performs the sum of products for each element.

A friend function is a function that is not a member of a class but has the permission to access the private and protected members of the class. It is declared inside the class using the `friend` keyword but defined outside.

17. Write a program to implement Insertion Sort algorithm â–¼ 
// Insertion Sort Implementation
#include <iostream>
using namespace std;

// Function to perform insertion sort
void insertionSort(int arr[], int n) {
    int i, key, j;
    // Iterate from the second element (index 1) to the end
    for (i = 1; i < n; i++) {
        key = arr[i]; // Element to be inserted into the sorted subarray
        j = i - 1;

        // Move elements of arr[0..i-1], that are greater than key, 
        // to one position ahead of their current position
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j = j - 1;
        }
        arr[j + 1] = key; // Insert the key into its correct position
    }
}

// Function to display array
void displayArray(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arr[] = {12, 11, 13, 5, 6};
    int n = sizeof(arr) / sizeof(arr[0]);
    
    // Display original array
    cout << "Original array: ";
    displayArray(arr, n);
    
    // Sort the array
    insertionSort(arr, n);
    
    // Display sorted array
    cout << "Sorted array: ";
    displayArray(arr, n);
    
    return 0;
}

Explanation:

Insertion sort builds the final sorted array one item at a time. It iterates through the input elements and inserts each element into its proper place in the sorted part of the array.

20. Write a program to implement a simple exception handling mechanism for division by zero â–¼ 
// Exception Handling - Division by Zero
#include <iostream>
using namespace std;

// Function that performs division and throws an exception
double safeDivide(double numerator, double denominator) {
    if (denominator == 0.0) {
        // Throw an exception (here, a string literal)
        throw "Error: Division by zero is not allowed!";
    }
    return numerator / denominator;
}

int main() {
    double a = 10.0;
    double b = 2.0;
    double c = 0.0;
    
    // Successful division
    try {
        cout << "Attempting to divide " << a << " by " << b << endl;
        double result = safeDivide(a, b);
        cout << "Result: " << result << endl;
    }
    catch (const char* msg) {
        // This catch block will not execute
        cerr << msg << endl;
    }

    cout << "\n--------------------------\n" << endl;

    // Division by zero (Exception will be thrown)
    try {
        cout << "Attempting to divide " << a << " by " << c << endl;
        double result = safeDivide(a, c);
        cout << "Result: " << result << endl; // This line is skipped
    }
    catch (const char* msg) {
        // Catch block handles the thrown exception
        cerr << "Exception Caught: " << msg << endl;
    }
    
    cout << "\nProgram continues after exception handling." << endl;
    
    return 0;
}

Explanation:

The program uses `try`, `throw`, and `catch`. If the `safeDivide` function detects a division by zero, it uses `throw` to generate an exception (a string). The `catch` block intercepts this exception, preventing program crash and allowing graceful error handling.
