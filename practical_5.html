#include <iostream>
using namespace std;

class Distance {
private:
    int feet;
    float inches;
    static const float MTF;  // Meter to Feet conversion (3.28)
    
public:
    // Default Constructor - feet ‡∑É‡∑Ñ inches 0 ‡∂ß initialize ‡∂ö‡∂ª‡∂∫‡∑í
    Distance() {
        feet = 0;
        inches = 0.0;
        cout << "Default Distance ‡∂±‡∑í‡∂ª‡∑ä‡∂∏‡∑è‡∂´‡∂∫ ‡∂ö‡∂ª‡∂± ‡∂Ω‡∂Ø‡∑ì" << endl;
    }
    
    // Meter to Distance Conversion Constructor
    Distance(float meter) {
        cout << "\nMeters ‡∑É‡∑í‡∂ß Distance ‡∑Ä‡∑ô‡∂≠ convert ‡∂ö‡∂ª‡∂∏‡∑í‡∂±‡∑ä..." << endl;
        cout << "Input: " << meter << " meters" << endl;
        
        float f = MTF * meter;  // ‡∂∏‡∑î‡∑Ö‡∑î feet (decimal ‡∑É‡∂∏‡∂ü)
        feet = static_cast(f);  // ‡∂¥‡∑ñ‡∂ª‡∑ä‡∂´ ‡∂Ü‡∂ö‡∑è‡∂ª‡∂∫‡∑ö feet
        inches = 12 * (f - feet);  // ‡∂â‡∂≠‡∑í‡∂ª‡∑í ‡∂ö‡∑ú‡∂ß‡∑É inches ‡∂∂‡∑Ä‡∂ß
        
        cout << "Converted: " << feet << " feet, " << inches << " inches" << endl;
    }
    
    // Parameterized Constructor
    Distance(int ft, float in) {
        feet = ft;
        inches = in;
    }
    
    // Display function (const function)
    void showdist() const {
        cout << "Distance: " << feet << " feet, " << inches << " inches" << endl;
    }
    
    // Distance to Meter Conversion Operator
    operator float() const {
        float f = inches / 12.0;  // inches ‡∂ë‡∂ö feet ‡∂∂‡∑Ä‡∂ß
        f += static_cast(feet);  // ‡∂∏‡∑î‡∑Ö‡∑î feet ‡∂ë‡∂ö‡∂≠‡∑î ‡∂ö‡∂ª‡∂±‡∑ä‡∂±
        return f / MTF;  // feet ‡∂ë‡∂ö meters ‡∂∂‡∑Ä‡∂ß
    }
};
