Enumerators, Scope Resolution Operator & Static Data Members

Enumerators (enum)

Enumerator යනු user-defined data type එකකි. එය නම් කළ integer constants සමූහයක් නිර්වචනය කිරීමට භාවිතා වේ. Enumerations කේතය වඩාත් කියවීමට පහසු සහ නඩත්තු කිරීමට පහසු කරයි.

මූලික Syntax

enum EnumName {
    CONSTANT1,
    CONSTANT2,
    CONSTANT3
};

සරල උදාහරණය

Basic Enum 
#include <iostream>
using namespace std;

enum Color {
    RED,      // 0
    GREEN,    // 1
    BLUE,     // 2
    YELLOW    // 3
};

int main() {
    Color myColor = GREEN;
    
    cout << "My color value: " << myColor << endl;
    
    if (myColor == GREEN) {
        cout << "The color is GREEN!" << endl;
    }
    
    return 0;
}
Output:
My color value: 1
The color is GREEN!

අගයන් ලබා දීම (Custom Values)

Enum constants වලට ඔබට අවශ්‍ය අගයන් ලබා දිය හැක:

Enum with Custom Values 
#include <iostream>
using namespace std;

enum Days {
    MONDAY = 1,
    TUESDAY = 2,
    WEDNESDAY = 3,
    THURSDAY = 4,
    FRIDAY = 5,
    SATURDAY = 6,
    SUNDAY = 7
};

enum Months {
    JAN = 1,
    FEB,      // 2 (automatically)
    MAR,      // 3
    APR = 10,
    MAY,      // 11
    JUN       // 12
};

int main() {
    Days today = WEDNESDAY;
    Months currentMonth = MAY;
    
    cout << "Day: " << today << endl;
    cout << "Month: " << currentMonth << endl;
    
    return 0;
}
Output:
Day: 3
Month: 11

Enum Class (C++11 සහ ඉහළ)

Enum class (scoped enum) භාවිතයෙන් වඩාත් type-safe සහ scope කළ enumerations නිර්මාණය කළ හැක:

Enum Class 
#include <iostream>
using namespace std;

enum class TrafficLight {
    RED,
    YELLOW,
    GREEN
};

enum class Status {
    ACTIVE,
    INACTIVE,
    PENDING
};

int main() {
    TrafficLight light = TrafficLight::RED;
    Status userStatus = Status::ACTIVE;
    
    if (light == TrafficLight::RED) {
        cout << "Stop!" << endl;
    }
    
    if (userStatus == Status::ACTIVE) {
        cout << "User is active" << endl;
    }
    
    return 0;
}
Enum vs Enum Class
  • enum: Values global scope එකට leak වේ
  • enum class: Values scoped වන අතර type-safe
  • enum class භාවිතා කිරීම නවීන C++ හි recommended වේ

Scope Resolution Operator (::)

Scope Resolution Operator (::) යනු C++ හි විශේෂ operator එකකි. එය classes, namespaces, සහ enums හි members වෙත ප්‍රවේශ වීමට භාවිතා වේ. මෙය විවිධ scopes හි එකම නමකින් පවතින identifiers අතර වෙනස හඳුනා ගැනීමට උපකාරී වේ.

භාවිත අවස්ථා

1. Class හි Member Functions Define කිරීම
Define Member Functions 
#include <iostream>
using namespace std;

class Rectangle {
private:
    int length;
    int width;
    
public:
    void setDimensions(int l, int w);
    int getArea();
};

// Scope resolution operator භාවිතා කරමින් functions define කිරීම
void Rectangle::setDimensions(int l, int w) {
    length = l;
    width = w;
}

int Rectangle::getArea() {
    return length * width;
}

int main() {
    Rectangle rect;
    rect.setDimensions(10, 5);
    cout << "Area: " << rect.getArea() << endl;
    return 0;
}
2. Global සහ Local Variables අතර වෙනස හඳුනා ගැනීම
Global vs Local 
#include <iostream>
using namespace std;

int value = 100;  // Global variable

int main() {
    int value = 50;  // Local variable
    
    cout << "Local value: " << value << endl;
    cout << "Global value: " << ::value << endl;  // :: භාවිතයෙන් global variable access කිරීම
    
    return 0;
}
Output:
Local value: 50
Global value: 100
3. Namespace Members වෙත ප්‍රවේශය
Namespace Access 
#include <iostream>
using namespace std;

namespace First {
    int x = 10;
    void display() {
        cout << "First namespace: " << x << endl;
    }
}

namespace Second {
    int x = 20;
    void display() {
        cout << "Second namespace: " << x << endl;
    }
}

int main() {
    First::display();   // First namespace එකේ function
    Second::display();  // Second namespace එකේ function
    
    cout << "First::x = " << First::x << endl;
    cout << "Second::x = " << Second::x << endl;
    
    return 0;
}
Output:
First namespace: 10
Second namespace: 20
First::x = 10
Second::x = 20
4. Nested Classes වෙත ප්‍රවේශය
Nested Classes 
#include <iostream>
using namespace std;

class Outer {
public:
    class Inner {
    public:
        void display() {
            cout << "Inside Inner class" << endl;
        }
    };
};

int main() {
    Outer::Inner obj;
    obj.display();
    return 0;
}

Static Data Members

Static data members යනු class එකට අයත් වන විශේෂ members වේ. මේවා class එකේ සියලුම objects අතර බෙදා ගනු ලබන අතර, එක් copy එකක් පමණක් memory හි පවතී. Static members class name භාවිතයෙන් access කළ හැක.

Static Data Member උදාහරණය

Static Variable 
#include <iostream>
using namespace std;

class Student {
private:
    int rollNo;
    static int count;  // Static data member declaration
    
public:
    Student() {
        rollNo = ++count;
    }
    
    void showData() {
        cout << "Roll No: " << rollNo << endl;
    }
    
    static void showCount() {  // Static member function
        cout << "Total students: " << count << endl;
    }
};

// Static data member initialization (class බාහිරව)
int Student::count = 0;

int main() {
    Student s1, s2, s3;
    
    s1.showData();
    s2.showData();
    s3.showData();
    
    // Static function call - object එකක් නොමැතිව
    Student::showCount();
    
    return 0;
}
Output:
Roll No: 1
Roll No: 2
Roll No: 3
Total students: 3
වැදගත් සටහන්
  • Static data members class definition එකේ declare කළත්, initialize කළ යුත්තේ class එකෙන් බාහිරව
  • Static members constructor මගින් initialize නොවේ
  • Static member functions හට non-static members වෙත සෘජුවම ප්‍රවේශය නැත

Static Member Functions

Static member functions යනු class එකට අයත් වන විශේෂ functions වේ. මේවාට object එකක් නොමැතිව call කළ හැක:

Static Functions 
#include <iostream>
using namespace std;

class Counter {
private:
    static int objectCount;
    
public:
    Counter() {
        objectCount++;
    }
    
    ~Counter() {
        objectCount--;
    }
    
    static int getCount() {
        return objectCount;
    }
};

int Counter::objectCount = 0;

int main() {
    cout << "Initial count: " << Counter::getCount() << endl;
    
    Counter c1, c2;
    cout << "After creating 2 objects: " << Counter::getCount() << endl;
    
    {
        Counter c3, c4;
        cout << "After creating 2 more: " << Counter::getCount() << endl;
    }
    
    cout << "After inner scope ends: " << Counter::getCount() << endl;
    
    return 0;
}
Output:
Initial count: 0
After creating 2 objects: 2
After creating 2 more: 4
After inner scope ends: 2

Practical Example: Bank Account

Real-World Example 
#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    string accountHolder;
    double balance;
    static double interestRate;  // සියලු accounts වලට පොදු
    static int totalAccounts;
    
public:
    BankAccount(string name, double bal) {
        accountHolder = name;
        balance = bal;
        totalAccounts++;
    }
    
    void applyInterest() {
        balance += balance * interestRate / 100;
    }
    
    void display() {
        cout << "Account: " << accountHolder 
             << ", Balance: Rs." << balance << endl;
    }
    
    static void setInterestRate(double rate) {
        interestRate = rate;
    }
    
    static void showInfo() {
        cout << "Total Accounts: " << totalAccounts << endl;
        cout << "Interest Rate: " << interestRate << "%" << endl;
    }
};

double BankAccount::interestRate = 5.0;
int BankAccount::totalAccounts = 0;

int main() {
    BankAccount acc1("Kamal", 10000);
    BankAccount acc2("Nimal", 15000);
    
    cout << "Initial state:" << endl;
    BankAccount::showInfo();
    cout << endl;
    
    acc1.display();
    acc2.display();
    cout << endl;
    
    // සියලු accounts සඳහා interest rate වෙනස් කිරීම
    BankAccount::setInterestRate(7.5);
    
    acc1.applyInterest();
    acc2.applyInterest();
    
    cout << "After applying interest:" << endl;
    BankAccount::showInfo();
    cout << endl;
    
    acc1.display();
    acc2.display();
    
    return 0;
}
