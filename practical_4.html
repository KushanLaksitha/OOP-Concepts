#include <iostream>

class Circle {

    double radius;
    
public:
    // Default Constructor - radius ‡∂ë‡∂ö 1.0 ‡∂ß set ‡∂ö‡∂ª‡∂∫‡∑í
    Circle() {
        radius = 1.0;
        cout << "Default ‡∑Ä‡∑ò‡∂≠‡∑ä‡∂≠‡∂∫‡∂ö‡∑ä ‡∂±‡∑í‡∂ª‡∑ä‡∂∏‡∑è‡∂´‡∂∫ ‡∂ö‡∂ª‡∂± ‡∂Ω‡∂Ø‡∑ì (radius = 1.0)" << endl;
    }
    
    // Parameterized Constructor - user input radius ‡∂∑‡∑è‡∑Ä‡∑í‡∂≠‡∑è ‡∂ö‡∂ª‡∂∫‡∑í
    Circle(double r) {
        radius = r;
        cout << "‡∑Ä‡∑ò‡∂≠‡∑ä‡∂≠‡∂∫‡∂ö‡∑ä ‡∂±‡∑í‡∂ª‡∑ä‡∂∏‡∑è‡∂´‡∂∫ ‡∂ö‡∂ª‡∂± ‡∂Ω‡∂Ø‡∑ì (radius = " << radius << ")" << endl;
    }
    
    // ‡∑Ä‡∂ª‡∑ä‡∂ú‡∂µ‡∂Ω‡∂∫ ‡∂ú‡∂´‡∂±‡∂∫ ‡∂ö‡∂ª‡∂± method - œÄ * r¬≤
    double area() {

    }
    
    // Radius ‡∂ë‡∂ö display ‡∂ö‡∑í‡∂ª‡∑ì‡∂∏
    void displayRadius() {

    }
