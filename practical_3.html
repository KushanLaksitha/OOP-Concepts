ප්‍රශ්නය 1: වෘත්තයක් නියෝජනය කරන Circle Class එක

මෙම වැඩසටහනෙන් වෘත්තයක් (Circle) නියෝජනය කරන class එකක් නිර්මාණය කරනවා. radius එක set කිරීමට, වර්ගඵලය ගණනය කිරීමට සහ display කිරීමට functions තුනක් ඇතුළත් වෙනවා. 
// Question 1: Circle Class
#include <iostream>
using namespace std;

class Circle {
private:
    double radius;

public:
    // Function to set radius
    void setRadius(double r) {
        radius = r;
    }

    // Function to calculate area
    double calculateArea() {
        return 3.14159 * radius * radius;
    }

    // Function to display area
    void displayArea() {
        cout << "වෘත්තයේ වර්ගඵලය: " << calculateArea() << " වර්ග ඒකක" << endl;
    }
};

int main() {
    Circle c;
    double r;

    cout << "වෘත්තයේ අරය (radius) ඇතුලත් කරන්න: ";
    cin >> r;

    c.setRadius(r);
    c.displayArea();

    return 0;
}

ප්‍රශ්නය 2: ගිටාර් වෙළඳසැල් පද්ධතිය

සංගීත උපකරණ වෙළඳසැලක් විසින් ගිටාර් විකිණීම සඳහා පරිගණක පද්ධතියක් භාවිතා කරනවා. ගිටාර් code එකක් ඇතුලත් කිරීමෙන් එම ගිටාරය පිළිබඳ විස්තර ලබා ගත හැකියි. වෙළඳසැලේ ගිටාර් වර්ග හතරක් පවතී. 
// Question 2: Guitar Shop System
#include <iostream>
#include <string>
using namespace std;

class Guitar {
private:
    int code;
    string brand;
    string model;
    double price;
    string type;

public:
    // Function to set guitar details
    void setDetails(int c, string b, string m, double p, string t) {
        code = c;
        brand = b;
        model = m;
        price = p;
        type = t;
    }

    // Function to get code
    int getCode() {
        return code;
    }

    // Function to display guitar details
    void displayDetails() {
        cout << "\n===== ගිටාර් විස්තර =====" << endl;
        cout << "කේතය: " << code << endl;
        cout << "සන්නාමය: " << brand << endl;
        cout << "ආකෘතිය: " << model << endl;
        cout << "මිල: Rs. " << price << endl;
        cout << "වර්ගය: " << type << endl;
        cout << "========================\n" << endl;
    }
};

int main() {
    Guitar guitars[4];
    
    // Initialize guitars with details
    guitars[0].setDetails(101, "Fender", "Stratocaster", 125000.00, "Electric");
    guitars[1].setDetails(102, "Gibson", "Les Paul", 185000.00, "Electric");
    guitars[2].setDetails(103, "Yamaha", "F310", 35000.00, "Acoustic");
    guitars[3].setDetails(104, "Taylor", "214ce", 275000.00, "Acoustic");

    int searchCode;
    bool found = false;

    cout << "ගිටාර් කේතය ඇතුලත් කරන්න: ";
    cin >> searchCode;

    // Search for guitar
    for(int i = 0; i < 4; i++) {
        if(guitars[i].getCode() == searchCode) {
            guitars[i].displayDetails();
            found = true;
            break;
        }
    }

    if(!found) {
        cout << "මෙම කේතය සඳහා ගිටාර් එකක් හමු නොවීය!" << endl;
    }

    return 0;
}

ප්‍රශ්නය 3: බැංකු ගිණුම් පද්ධතිය

බැංකු ගිණුමක් නියෝජනය කරන class එකක් නිර්මාණය කරනවා. ගිණුම් හිමියාගේ නම, ගිණුම් අංකය සහ ශේෂය data members ලෙස ඇතුළත් වෙනවා. මුදල් තැන්පත් කිරීමට සහ ආපසු ගැනීමට functions ඇතුළත් වෙනවා. 
// Question 3: Bank Account System
#include <iostream>
#include <string>
using namespace std;

class BankAccount {
private:
    string accountHolderName;
    long long accountNumber;
    double balance;

public:
    // Function to set account details
    void setAccountDetails(string name, long long accNum, double bal) {
        accountHolderName = name;
        accountNumber = accNum;
        balance = bal;
    }

    // Function to withdraw money
    void withdraw(double amount) {
        if(amount > balance) {
            cout << "ප්‍රමාණවත් මුදල් නොමැත!" << endl;
            cout << "වත්මන් ශේෂය: Rs. " << balance << endl;
        } else {
            balance -= amount;
            cout << "Rs. " << amount << " සාර්ථකව ආපසු ගන්නා ලදී." << endl;
            cout << "නව ශේෂය: Rs. " << balance << endl;
        }
    }

    // Function to deposit money
    void deposit(double amount) {
        balance += amount;
        cout << "Rs. " << amount << " සාර්ථකව තැන්පත් කරන ලදී." << endl;
        cout << "නව ශේෂය: Rs. " << balance << endl;
    }

    // Function to display account details
    void displayData() {
        cout << "\n===== ගිණුම් විස්තර =====" << endl;
        cout << "ගිණුම් හිමියා: " << accountHolderName << endl;
        cout << "ගිණුම් අංකය: " << accountNumber << endl;
        cout << "ශේෂය: Rs. " << balance << endl;
        cout << "======================\n" << endl;
    }
};

int main() {
    BankAccount account;
    string name;
    long long accNum;
    double initialBalance, amount;
    int choice;

    cout << "ගිණුම් හිමියාගේ නම ඇතුලත් කරන්න: ";
    cin.ignore();
    getline(cin, name);

    cout << "ගිණුම් අංකය ඇතුලත් කරන්න: ";
    cin >> accNum;

    cout << "ආරම්භක ශේෂය ඇතුලත් කරන්න: Rs. ";
    cin >> initialBalance;

    account.setAccountDetails(name, accNum, initialBalance);
    account.displayData();

    do {
        cout << "\n----- මෙනුව -----" << endl;
        cout << "1. මුදල් තැන්පත් කරන්න" << endl;
        cout << "2. මුදල් ආපසු ගන්න" << endl;
        cout << "3. ගිණුම් විස්තර පෙන්වන්න" << endl;
        cout << "4. පිටවන්න" << endl;
        cout << "ඔබේ තේරීම: ";
        cin >> choice;

        switch(choice) {
            case 1:
                cout << "තැන්පත් කිරීමේ මුදල: Rs. ";
                cin >> amount;
                account.deposit(amount);
                break;
            case 2:
                cout << "ආපසු ගැනීමේ මුදල: Rs. ";
                cin >> amount;
                account.withdraw(amount);
                break;
            case 3:
                account.displayData();
                break;
            case 4:
                cout << "ස්තූතියි!" << endl;
                break;
            default:
                cout << "වැරදි තේරීමක්!" << endl;
        }
    } while(choice != 4);

    return 0;
}

ගෙදර වැඩ පැවරුම: ශිෂ්‍ය ශ්‍රේණිගත කිරීමේ පද්ධතිය

ශිෂ්‍යන්ගේ ලකුණු ඇතුළත් කර ශ්‍රේණිය (Grade) ගණනය කරන පද්ධතියක් නිර්මාණය කරනවා. මෙහි ශිෂ්‍ය විස්තර, විෂයයන් සහ ලකුණු ඇතුළත් වෙනවා. 
// Take Home Assignment: Student Grading System
#include <iostream>
#include <string>
using namespace std;

class Student {
private:
    string name;
    int studentID;
    double marks[5];
    double average;
    string grade;

public:
    // Function to set student details
    void setStudentDetails(string n, int id) {
        name = n;
        studentID = id;
    }

    // Function to input marks
    void inputMarks() {
        cout << "\nවිෂයයන් 5 සඳහා ලකුණු ඇතුලත් කරන්න:\n";
        for(int i = 0; i < 5; i++) {
            cout << "විෂය " << (i+1) << ": ";
            cin >> marks[i];
            
            // Validation
            while(marks[i] < 0 || marks[i] > 100) {
                cout << "වලංගු නොවේ! නැවත ඇතුලත් කරන්න (0-100): ";
                cin >> marks[i];
            }
        }
    }

    // Function to calculate average
    void calculateAverage() {
        double sum = 0;
        for(int i = 0; i < 5; i++) {
            sum += marks[i];
        }
        average = sum / 5.0;
    }

    // Function to determine grade
    void determineGrade() {
        if(average >= 75) {
            grade = "A (විශිෂ්ට)";
        } else if(average >= 65) {
            grade = "B (ඉතා හොඳ)";
        } else if(average >= 55) {
            grade = "C (හොඳ)";
        } else if(average >= 35) {
            grade = "S (සමත්)";
        } else {
            grade = "F (අසමත්)";
        }
    }

    // Function to display student details
    void displayDetails() {
        cout << "\n========== ශිෂ්‍ය වාර්තාව ==========" << endl;
        cout << "නම: " << name << endl;
        cout << "අංකය: " << studentID << endl;
        cout << "\nවිෂය ලකුණු:" << endl;
        for(int i = 0; i < 5; i++) {
            cout << "  විෂය " << (i+1) << ": " << marks[i] << endl;
        }
        cout << "\nසාමාන්‍ය ලකුණු: " << average << endl;
        cout << "ශ්‍රේණිය: " << grade << endl;
        cout << "===================================\n" << endl;
    }
};

int main() {
    Student student;
    string name;
    int id;

    cout << "===== ශිෂ්‍ය ශ්‍රේණිගත කිරීමේ පද්ධතිය =====\n" << endl;
    
    cout << "ශිෂ්‍ය නම ඇතුලත් කරන්න: ";
    cin.ignore();
    getline(cin, name);

    cout << "ශිෂ්‍ය අංකය ඇතුලත් කරන්න: ";
    cin >> id;

    student.setStudentDetails(name, id);
    student.inputMarks();
    student.calculateAverage();
    student.determineGrade();
    student.displayDetails();

    return 0;
}
