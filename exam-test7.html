01. Class Numbers: Find Max and Min in a Fixed Array ‚ñº 
// 01. Class Numbers: Find Max/Min
#include <iostream>
include <algorithm> // For min_element and max_element (used for concise implementation)
using namespace std;

class Numbers {
private:
    static const int SIZE = 10;
    int arr[SIZE];

public:
    // Function to initialize the array (setter)
    void setArray(const int values[SIZE]) {
        for (int i = 0; i < SIZE; i++) {
            arr[i] = values[i];
        }
    }

    // Returns the largest number in the array
    int findMax() const {
        int maxVal = arr[0];
        for (int i = 1; i < SIZE; i++) {
            if (arr[i] > maxVal) {
                maxVal = arr[i];
            }
        }
        return maxVal;
    }

    // Returns the smallest number in the array
    int findMin() const {
        int minVal = arr[0];
        for (int i = 1; i < SIZE; i++) {
            if (arr[i] < minVal) {
                minVal = arr[i];
            }
        }
        return minVal;
    }

    // Prints both the maximum and minimum values
    void display() const {
        cout << "Maximum Value: " << findMax() << endl;
        cout << "Minimum Value: " << findMin() << endl;
    }
};

int main() {
    // Create an object of Numbers.
    Numbers myNumbers;
    
    // Initialize the array with the given values.
    int initialValues[10] = {25, 7, 19, 42, 13, 5, 60, 17, 9, 30};
    myNumbers.setArray(initialValues);
    
    // Call the display() function.
    myNumbers.display();
    
    return 0;
}

Explanation:

The Numbers class encapsulates the fixed-size array. The findMax() and findMin() methods iterate through the array to find the extreme values in a single pass (O(n)). The display() function calls these two methods to produce the required output, correctly demonstrating encapsulation and member function usage.

02. Sum of Positive and Negative Numbers in an Array ‚ñº 
// 02. Sum of Positive and Negative Numbers
#include <iostream>
using namespace std;

// Function that accepts an integer array and its size
void calculateSumTypes(const int arr[], int size) {
    int sumPositive = 0;
    int sumNegative = 0;
    
    for (int i = 0; i < size; i++) {
        if (arr[i] > 0) {
            sumPositive += arr[i];
        } else if (arr[i] < 0) {
            sumNegative += arr[i];
        }
        // Note: The '0' element is ignored as required by the logic (positive/negative)
    }
    
    cout << "Sum of positive numbers: " << sumPositive << endl;
    cout << "Sum of negative numbers: " << sumNegative << endl;
}

int main() {
    // Create the specified integer array.
    int numbers[] = {10, -3, 25, -8, 15, -1, 0, -7, 12, 9};
    int size = sizeof(numbers) / sizeof(numbers[0]);
    
    // Pass it to the function.
    calculateSumTypes(numbers, size);
    
    return 0;
}

Explanation:

The calculateSumTypes() function iterates through the input array once. It uses simple conditional logic (if/else if) to determine if each element is positive or negative, adding it to the respective running sum. This approach correctly segregates and sums the numbers as required in a single pass (O(n)).

03. Abstract Base Class Shape and Derived Class Circle (Pure Virtual Function) ‚ñº 
// 03. Abstract Class and Circle Derived Class
#include <iostream>
include <cmath> // For PI constant (M_PI) or define a constant
include <iomanip> // For setprecision
using namespace std;

// Base Class (Abstract) - Provided in the question
class Shape {
public:
    // Pure virtual function: makes Shape an abstract class
    virtual double calculateArea() = 0;
    
    virtual void display() {
        cout << "This is a shape" << endl;
    }
    virtual ~Shape() {} // Virtual destructor for safety
};

// Concrete Derived Class
class Circle : public Shape {
private:
    double radius;
    const double PI = 3.14159265358979323846; // Define a high precision PI

public:
    // Constructor that takes radius as a parameter
    Circle(double r) : radius(r) {}
    
    // Implementation of the pure virtual function
    double calculateArea() override {
        return PI * radius * radius;
    }

    // Override display() function to print details
    void display() override {
        cout << fixed << setprecision(2);
        cout << "Radius: " << radius << endl;
        cout << "Area: " << calculateArea() << endl;
    }
};

int main() {
    // Create an object of the Circle class with radius = 7.
    Circle myCircle(7.0);
    
    // Call the display() function to show the details.
    myCircle.display();
    
    return 0;
}

Explanation:

The Shape class is an Abstract Base Class due to the presence of the pure virtual function calculateArea() = 0;. The derived Circle class is a **concrete** class because it fully implements all inherited pure virtual functions. It includes a constructor for initialization and overrides the display() and calculateArea() functions to provide circle-specific logic and output.

04. Abstract Base Class Vehicle and Derived Class Car (Function Overriding) ‚ñº 
// 04. Vehicle Base Class and Car Derived Class
#include <iostream>
using namespace std;

// Base Class - Provided in the question
class Vehicle {
public:
    // virtual ensures start() can be overridden
    virtual void start() {
        cout << "Vehicle starts" << endl;
    }
    // Pure virtual function
    virtual void fuelType() = 0; 
    virtual ~Vehicle() {}
};

// Concrete Derived Class
class Car : public Vehicle {
public:
    // Override start() to print "Car is starting with a key ignition."
    void start() override {
        cout << "Car is starting with a key ignition." << endl;
    }
    
    // Implement fuelType() to print "Fuel type: Petrol"
    void fuelType() override {
        cout << "Fuel type: Petrol" << endl;
    }
};

int main() {
    // Create a Car object
    Car myCar;
    
    // Call both methods.
    myCar.start();
    myCar.fuelType();
    
    return 0;
}

Explanation:

The Car class inherits from the abstract Vehicle class. It must implement the pure virtual function fuelType() to become a concrete class. It also overrides the base class's non-pure virtual function start(). When myCar.start() and myCar.fuelType() are called, the Car-specific implementations are executed, as intended by polymorphism and overriding.

05. Count Uppercase and Lowercase Letters in a String ‚ñº 
// 05. Count Uppercase and Lowercase Letters
#include <iostream>
include <string>
include <cctype> // For isupper, islower
using namespace std;

void countCases(const string& str) {
    int uppercaseCount = 0;
    int lowercaseCount = 0;

    // Iterate through each character of the string
    for (char ch : str) {
        // Check if the character is uppercase
        if (isupper(ch)) {
            uppercaseCount++;
        } 
        // Check if the character is lowercase
        else if (islower(ch)) {
            lowercaseCount++;
        }
        // Digits, spaces, and special characters are ignored by design
    }

    cout << "Number of uppercase letters: " << uppercaseCount << endl;
    cout << "Number of lowercase letters: " << lowercaseCount << endl;
}

int main() {
    string inputString;
    
    // Take a string input from the user.
    cout << "Enter a string: ";
    // Use getline to read the entire line, including spaces
    getline(cin, inputString);
    
    // Count and print both counts.
    countCases(inputString);
    
    return 0;
}
/* 
Example Execution:

Expected Output:
Number of uppercase letters: 4
Number of lowercase letters: 12
*/

Explanation:

The program uses the `<cctype>` library functions isupper() and islower(). It iterates through the input string using a range-based for loop. Only characters identified as letters (upper or lower case) are counted, effectively ignoring digits, spaces, and other symbols as requested.

// 06. Student Class: Calculate Average Marks
#include <iostream>
include <string>
include <numeric> // For std::accumulate (for concise sum)
include <iomanip> // For setprecision
using namespace std;

class Student {
private:
    string name;
    int marks[5];
    static const int NUM_SUBJECTS = 5;

public:
    // Constructor to initialize the student's name and marks.
    Student(string n, const int m[NUM_SUBJECTS]) : name(n) {
        for (int i = 0; i < NUM_SUBJECTS; i++) {
            marks[i] = m[i];
        }
    }

    // Function that returns the average of marks.
    double calculateAverage() const {
        int sum = 0;
        for (int mark : marks) {
            sum += mark;
        }
        return (double)sum / NUM_SUBJECTS;
    }

    // Function that shows the name and average marks.
    void display() const {
        cout << fixed << setprecision(1);
        cout << "Student Name: " << name << endl;
        cout << "Average Marks: " << calculateAverage() << endl;
    }
};

int main() {
    // Create a Student object named "Nimal" with marks {80, 75, 90, 85, 88}.
    int nimalMarks[5] = {80, 75, 90, 85, 88};
    Student nimal("Nimal", nimalMarks);
    
    // Display the name and average marks.
    nimal.display();
    
    return 0;
}

Explanation:

The Student class uses a constructor to initialize both the name and the array of marks. The calculateAverage() method sums all the marks and divides by the fixed number of subjects (5). A type cast to double is used during the division to ensure floating-point precision is maintained, correctly matching the required output format (83.6).
