Static vs Dynamic Binding

Compile-time සහ Runtime Binding පිළිබඳ සම්පූර්ණ විස්තරය

Binding යනු කුමක්ද?

Binding යනු function call එකක් එහි definition එක සමඟ සම්බන්ධ කිරීමයි. C++ හි Binding වර්ග දෙකකි:

Static Binding

Early Binding හෝ Compile-time Binding ලෙසද හැඳින්වේ

Compiler විසින් compile time එකේදී function call එක resolve කරනු ලැබේ

Dynamic Binding

Late Binding හෝ Runtime Binding ලෙසද හැඳින්වේ

Program run වන අතරතුරදී (runtime) function call එක resolve කරනු ලැබේ

Static Binding

Static Binding හි function call එක compile time එකේදී resolve වේ. මෙහිදී compiler විසින් කුමන function එක call කරන්නේද යන්න දැනටමත් දනී.

ප්‍රධාන ලක්ෂණ:
  • වේගවත් performance (කෙලින්ම function එකට යනවා)
  • Compile time එකේදී decide වේ
  • Normal functions, function overloading සහ operator overloading භාවිතා කරයි
  • Virtual keyword භාවිතා නොකරයි

Static Binding උදාහරණය:

C++ - Static Binding 
#include <iostream>
using namespace std;

class Animal {
public:
    // Normal function - Static Binding
    void sound() {
        cout << "සතා හඬ නගයි" << endl;
    }
};

class Dog : public Animal {
public:
    // Override function - නමුත් virtual නැති නිසා Static Binding
    void sound() {
        cout << "බුරා බුරා!" << endl;
    }
};

int main() {
    Animal* ptr;  // Base class pointer
    Dog myDog;
    
    ptr = &myDog;  // Pointer එක Dog object එකට point කරනවා
    
    // Static Binding - Compile time එකේදී decide වේ
    // Base class function එක call වේ (pointer type එකට අනුව)
    ptr->sound();  // Output: සතා හඬ නගයි
    
    // Object එකෙන් කෙලින්ම call කරනවා නම්
    myDog.sound(); // Output: බුරා බුරා!
    
    return 0;
}
සටහන: Base class pointer භාවිතා කරන විට, pointer එකේ type එකට අනුව function එක call වේ (object type එකට අනුව නොව)

Dynamic Binding

Dynamic Binding හි function call එක runtime එකේදී resolve වේ. මෙය සිදු කිරීමට virtual keyword භාවිතා කරනු ලැබේ.

ප්‍රධාන ලක්ෂණ:
  • Runtime polymorphism සාක්ෂාත් කරයි
  • Runtime එකේදී decide වේ
  • virtual keyword භාවිතා කරයි
  • Object එකේ actual type එකට අනුව function එක call වේ
  • Virtual table (vtable) භාවිතා කරයි

Dynamic Binding උදාහරණය:

C++ - Dynamic Binding 
#include <iostream>
using namespace std;

class Animal {
public:
    // Virtual function - Dynamic Binding සඳහා
    virtual void sound() {
        cout << "සතා හඬ නගයි" << endl;
    }
    
    virtual void eat() {
        cout << "සතා කනවා" << endl;
    }
    
    // Virtual destructor - Important!
    virtual ~Animal() {
        cout << "Animal destructor" << endl;
    }
};

class Dog : public Animal {
public:
    // Override virtual function
    void sound() override {
        cout << "බුරා බුරා!" << endl;
    }
    
    void eat() override {
        cout << "බල්ලා කෑම කනවා" << endl;
    }
    
    ~Dog() {
        cout << "Dog destructor" << endl;
    }
};

class Cat : public Animal {
public:
    void sound() override {
        cout << "මියාව් මියාව්!" << endl;
    }
    
    void eat() override {
        cout << "බළලා කෑම කනවා" << endl;
    }
    
    ~Cat() {
        cout << "Cat destructor" << endl;
    }
};

int main() {
    Animal* ptr;  // Base class pointer
    
    // Dog object එකක් point කරනවා
    Dog myDog;
    ptr = &myDog;
    
    // Dynamic Binding - Runtime එකේදී Dog version call වේ
    cout << "Dog හඬ: ";
    ptr->sound();  // Output: බුරා බුරා!
    ptr->eat();    // Output: බල්ලා කෑම කනවා
    
    cout << "\n";
    
    // Cat object එකක් point කරනවා
    Cat myCat;
    ptr = &myCat;
    
    // Dynamic Binding - Runtime එකේදී Cat version call වේ
    cout << "Cat හඬ: ";
    ptr->sound();  // Output: මියාව් මියාව්!
    ptr->eat();    // Output: බළලා කෑම කනවා
    
    return 0;
}
වාසි: Pointer type එක කුමක් වුවත්, actual object type එකට අනුව correct function එක call වේ!

Shape hierarchy එකක් භාවිතයෙන් Static vs Dynamic Binding දෙකම බලමු:

C++ - Complete Example 
#include <iostream>
using namespace std;

// Base class
class Shape {
public:
    // Non-virtual function - Static Binding
    void display() {
        cout << "මෙය Shape එකක්" << endl;
    }
    
    // Virtual function - Dynamic Binding
    virtual void draw() {
        cout << "Shape එකක් draw කරනවා" << endl;
    }
    
    virtual double area() {
        return 0.0;
    }
};

// Derived class - Circle
class Circle : public Shape {
private:
    double radius;
    
public:
    Circle(double r) : radius(r) {}
    
    // Override non-virtual function
    void display() {
        cout << "මෙය Circle එකක්" << endl;
    }
    
    // Override virtual functions
    void draw() override {
        cout << "Circle එකක් draw කරනවා (රවුම)" << endl;
    }
    
    double area() override {
        return 3.14159 * radius * radius;
    }
};

// Derived class - Rectangle
class Rectangle : public Shape {
private:
    double width, height;
    
public:
    Rectangle(double w, double h) : width(w), height(h) {}
    
    void display() {
        cout << "මෙය Rectangle එකක්" << endl;
    }
    
    void draw() override {
        cout << "Rectangle එකක් draw කරනවා (සෘජුකෝණාස්‍රය)" << endl;
    }
    
    double area() override {
        return width * height;
    }
};

int main() {
    Circle circle(5.0);
    Rectangle rect(4.0, 6.0);
    
    Shape* shape1 = &circle;
    Shape* shape2 = ▭
    
    cout << "=== Static Binding (display function) ===" << endl;
    // Non-virtual function - Static Binding
    // Pointer type අනුව call වේ
    shape1->display();  // Output: මෙය Shape එකක්
    shape2->display();  // Output: මෙය Shape එකක්
    
    cout << "\n=== Dynamic Binding (draw function) ===" << endl;
    // Virtual function - Dynamic Binding
    // Actual object type අනුව call වේ
    shape1->draw();     // Output: Circle එකක් draw කරනවා (රවුම)
    shape2->draw();     // Output: Rectangle එකක් draw කරනවා (සෘජුකෝණාස්‍රය)
    
    cout << "\n=== Dynamic Binding (area function) ===" << endl;
    cout << "Circle Area: " << shape1->area() << endl;      // Actual calculation
    cout << "Rectangle Area: " << shape2->area() << endl;   // Actual calculation
    
    cout << "\n=== Direct Object Calls ===" << endl;
    // Object එකෙන් කෙලින්ම call කරනවා නම් correct version එක call වේ
    circle.display();   // Output: මෙය Circle එකක්
    rect.display();     // Output: මෙය Rectangle එකක්
    
    return 0;
}
Output විස්තරය:
  • display() - Non-virtual නිසා base class version එක call වේ (Static Binding)
  • draw() සහ area() - Virtual නිසා derived class version එක call වේ (Dynamic Binding)

Virtual Table (vtable) යනු කුමක්ද?

Dynamic Binding ක්‍රියාත්මක කිරීම සඳහා C++ compiler විසින් virtual table (vtable) එකක් භාවිතා කරයි.

vtable ක්‍රියාකාරිත්වය:
  1. Virtual functions ඇති සෑම class එකක් සඳහාම vtable එකක් තනයි
  2. vtable එකේ virtual function pointers ගබඩා වේ
  3. සෑම object එකක්ම vptr (virtual pointer) එකක් තබා ගනී
  4. Runtime එකේදී vptr භාවිතයෙන් correct function එක සොයා ගනී
vtable Structure
Animal vtable:
├── sound() → Animal::sound()
├── eat()   → Animal::eat()

Dog vtable:
├── sound() → Dog::sound()      (Overridden)
├── eat()   → Dog::eat()        (Overridden)

Cat vtable:
├── sound() → Cat::sound()      (Overridden)
├── eat()   → Cat::eat()        (Overridden)

කවදා භාවිතා කරන්නද?

Static Binding භාවිතා කරන්න:
  • Performance critical code සඳහා
  • Function overloading සඳහා
  • Operator overloading සඳහා
  • Simple, fixed behavior සඳහා
  • Inheritance hierarchy නැති විට
Dynamic Binding භාවිතා කරන්න:
  • Polymorphic behavior අවශ්‍ය විට
  • Base class pointers භාවිතා කරන විට
  • Runtime decision making අවශ්‍ය විට
  • Virtual destructors අවශ්‍ය විට
  • Interface-based design සඳහා

වැදගත් නීති සහ සටහන්

Virtual Destructor අනිවාර්යයි!

Base class එකක් වශයෙන් භාවිතා කරන class එකකට virtual destructor එකක් තිබිය යුතුයි:

class Base {
public:
    virtual ~Base() {  // Virtual destructor
        cout << "Base destructor" << endl;
    }
};

class Derived : public Base {
public:
    ~Derived() {
        cout << "Derived destructor" << endl;
    }
};

// Virtual destructor නැතිව
Base* ptr = new Derived();
delete ptr;  // අනතුරුදායකයි! Derived destructor call නොවේ

// Virtual destructor සමඟ
Base* ptr = new Derived();
delete ptr;  // ආරක්ෂිතයි! දෙකම call වේ
අමතර සටහන්:
  • Static functions virtual විය නොහැක
  • Constructors virtual විය නොහැක
  • Inline functions virtual වුවත් inline නොවේ
  • Friend functions virtual විය නොහැක
  • Virtual functions inheritance hierarchy අවශ්‍ය කරයි

Pure Virtual Functions

Pure virtual functions යනු implementation නැති virtual functions වේ. මේවා abstract classes සෑදීමට භාවිතා කරයි:

C++ - Pure Virtual 
#include <iostream>
using namespace std;

// Abstract class - pure virtual function ඇත
class Animal {
public:
    // Pure virtual function (= 0)
    virtual void sound() = 0;
    virtual void move() = 0;
    
    // Normal virtual function
    virtual void eat() {
        cout << "සතා කනවා" << endl;
    }
    
    virtual ~Animal() {}
};

class Dog : public Animal {
public:
    // Pure virtual functions implement කළ යුතුයි
    void sound() override {
        cout << "බුරා බුරා!" << endl;
    }
    
    void move() override {
        cout << "බල්ලා දුවනවා" << endl;
    }
};

class Fish : public Animal {
public:
    void sound() override {
        cout << "මාළු හඬ නැත" << endl;
    }
    
    void move() override {
        cout << "මාළු පිහිනනවා" << endl;
    }
    
    void eat() override {
        cout << "මාළු කෑම කනවා" << endl;
    }
};

int main() {
    // Animal obj;  // Error! Abstract class instantiate කළ නොහැක
    
    Animal* animals[2];
    
    Dog dog;
    Fish fish;
    
    animals[0] = &dog;
    animals[1] = &fish;
    
    for(int i = 0; i < 2; i++) {
        animals[i]->sound();
        animals[i]->move();
        animals[i]->eat();
        cout << "---" << endl;
    }
    
    return 0;
}
සටහන:
  • Pure virtual function ඇති class එකක් abstract class එකක් වේ
  • Abstract classes instantiate කළ නොහැක
  • Derived classes හි pure virtual functions implement කළ යුතුයි

override Keyword

C++11 සිට override keyword භාවිතයෙන් function එකක් virtual function එකක් override කරන බව පැහැදිලිව දැක්විය හැක:

C++ - override keyword 
class Base {
public:
    virtual void display() {
        cout << "Base display" << endl;
    }
    
    virtual void show(int x) {
        cout << "Base show: " << x << endl;
    }
};

class Derived : public Base {
public:
    // Correct override
    void display() override {
        cout << "Derived display" << endl;
    }
    
    // Compile error - signature එක match නොවේ
    // void show(double x) override {  // Error!
    //     cout << "Derived show: " << x << endl;
    // }
    
    // Correct override
    void show(int x) override {
        cout << "Derived show: " << x << endl;
    }
};
override keyword වාසි:
  • Compile time එකේදී override errors හඳුනා ගනී
  • Code readability වැඩි කරයි
  • Accidental mistakes වළක්වයි
  • Modern C++ best practice එකක්

සාරාංශය

  • Static Binding: Compile time එකේදී function call resolve වේ, වේගවත්, virtual keyword නැත
  • Dynamic Binding: Runtime එකේදී function call resolve වේ, virtual keyword භාවිතා කරයි
  • Virtual functions භාවිතයෙන් runtime polymorphism ලබා ගත හැක
  • vtable සහ vptr භාවිතයෙන් dynamic binding ක්‍රියාත්මක වේ
  • Virtual destructor inheritance භාවිතා කරන විට අනිවාර්යයි
  • Pure virtual functions භාවිතයෙන් abstract classes සෑදිය හැක
  • override keyword භාවිතයෙන් compile time errors හඳුනා ගත හැක
  • Static binding වේගවත්, Dynamic binding වඩා flexible
