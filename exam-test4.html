#include <iostream>
using namespace std;

bool isPrime(int n) {
    if (n <= 1) {
        return false; // 0 and 1 are not prime
    }
    // Check for factors from 2 up to the square root of n
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            return false; // Found a factor, so it's not prime
        }
    }
    return true; // No factors found
}

int main() {
    int num;
    cout << "Enter a positive integer: ";
    cin >> num;
    
    if (isPrime(num)) {
        cout << num << " is a prime number." << endl;
    } else {
        cout << num << " is NOT a prime number." << endl;
    }
    
    return 0;
}

Explanation:

A prime number is only divisible by 1 and itself. The optimized check loops from 2 up to the square root of the number. If any number in this range divides the input with no remainder (n % i == 0), the number is not prime.

3. Reverse a string without using any built-in C++ reverse functions â–¼ 
// String Reversal using Swap
#include <iostream>
include <string>
using namespace std;

void reverseString(string& str) {
    int n = str.length();
    // Loop only up to half the length
    for (int i = 0; i < n / 2; i++) {
        // Swap characters at position i and n - 1 - i
        char temp = str[i];
        str[i] = str[n - 1 - i];
        str[n - 1 - i] = temp;
    }
}

int main() {
    string myString = "UniversityLevel";
    cout << "Original string: " << myString << endl;
    
    reverseString(myString);
    
    cout << "Reversed string: " << myString << endl;
    
    return 0;
}

Explanation:

This approach uses a two-pointer logic (implied by i and n - 1 - i) to swap characters. By looping only half the string's length, it ensures each pair of characters (from start and end) is swapped exactly once, efficiently reversing the string in-place.

4. Write a program to find the Transpose of a 3x3 Matrix â–¼ 
// Matrix Transpose (3x3)
#include <iostream>
using namespace std;

const int SIZE = 3;

void transposeMatrix(int matrix[SIZE][SIZE]) {
    // The transpose can be calculated in-place by swapping matrix[i][j] with matrix[j][i]
    // We only iterate through the upper triangle to avoid double swapping
    for (int i = 0; i < SIZE; i++) {
        for (int j = i + 1; j < SIZE; j++) {
            // Swap matrix[i][j] and matrix[j][i]
            int temp = matrix[i][j];
            matrix[i][j] = matrix[j][i];
            matrix[j][i] = temp;
        }
    }
}

void displayMatrix(const int matrix[SIZE][SIZE]) {
    for (int i = 0; i < SIZE; i++) {
        for (int j = 0; j < SIZE; j++) {
            cout << matrix[i][j] << "\t";
        }
        cout << endl;
    }
}

int main() {
    int matrix[SIZE][SIZE] = {
        {1, 2, 3},
        {4, 5, 6},
        {7, 8, 9}
    };
    
    cout << "Original Matrix:" << endl;
    displayMatrix(matrix);
    
    transposeMatrix(matrix);
    
    cout << "\nTransposed Matrix:" << endl;
    displayMatrix(matrix);
    
    return 0;
}

Explanation:

The transpose of a matrix $A$ is $A^T$ where $A^T[i][j] = A[j][i]$. For a square matrix, this can be done in-place by swapping the elements symmetrically across the main diagonal. The nested loops iterate only over the upper-triangle (j = i + 1) to ensure each pair is swapped only once.

// Array Min/Max
#include <iostream>
using namespace std;

void findMinMax(int arr[], int size) {
    if (size <= 0) {
        cout << "Array is empty or invalid size." << endl;
        return;
    }
    
    // Initialize min and max with the first element
    int minVal = arr[0];
    int maxVal = arr[0];
    
    // Iterate from the second element
    for (int i = 1; i < size; i++) {
        if (arr[i] < minVal) {
            minVal = arr[i];
        }
        if (arr[i] > maxVal) {
            maxVal = arr[i];
        }
    }
    
    cout << "Largest element: " << maxVal << endl;
    cout << "Smallest element: " << minVal << endl;
}

int main() {
    int numbers[] = {45, 12, 88, 7, 99, 33};
    int n = sizeof(numbers) / sizeof(numbers[0]);
    
    findMinMax(numbers, n);
    
    return 0;
}

Explanation:

The program initializes both minVal and maxVal to the first element of the array. It then traverses the rest of the array only once, comparing each element against the current minimum and maximum values and updating them as necessary. This is an O(n) approach.

// Palindrome Number Check
#include <iostream>
using namespace std;

bool isPalindromeNumber(int n) {
    if (n < 0) return false;
    
    int originalNum = n;
    int reversedNum = 0;
    
    // Reverse the number
    while (n > 0) {
        int digit = n % 10;
        reversedNum = reversedNum * 10 + digit;
        n /= 10;
    }
    
    // Compare the original number with the reversed number
    return originalNum == reversedNum;
}

int main() {
    int num1 = 121;
    int num2 = 12345;
    
    cout << num1 << " is a palindrome: " << (isPalindromeNumber(num1) ? "Yes" : "No") << endl;
    cout << num2 << " is a palindrome: " << (isPalindromeNumber(num2) ? "Yes" : "No") << endl;
    
    return 0;
}

Explanation:

A palindrome number reads the same backward as forward. The program iteratively extracts the last digit of the original number (using modulo 10) and uses it to build a new, reversed number. Finally, it compares the original number with the reversed number.

13. Implement a basic Stack data structure using an array (LIFO principle) â–¼ 
// Stack Implementation (Array-based)
#include <iostream>
using namespace std;

const int MAX_SIZE = 5;

class Stack {
private:
    int arr[MAX_SIZE];
    int top; // Index of the top element

public:
    // Constructor: Initialize top to -1 (empty stack)
    Stack() : top(-1) {} 
    
    bool isEmpty() const {
        return top == -1;
    }
    
    bool isFull() const {
        return top == MAX_SIZE - 1;
    }
    
    // Push operation (add element)
    void push(int value) {
        if (isFull()) {
            cout << "Stack Overflow: Cannot push " << value << endl;
            return;
        }
        arr[++top] = value;
        cout << "Pushed: " << value << endl;
    }
    
    // Pop operation (remove top element)
    int pop() {
        if (isEmpty()) {
            cout << "Stack Underflow" << endl;
            return -1; // Error code
        }
        return arr[top--];
    }
};

int main() {
    Stack s;
    
    s.push(10);
    s.push(20);
    
    cout << "Popped: " << s.pop() << endl;
    
    s.push(30);
    s.push(40);
    s.push(50);
    s.push(60); // Should cause overflow
    
    cout << "Popped: " << s.pop() << endl;
    
    return 0;
}

Explanation:

The stack is a Last In, First Out (LIFO) data structure. The array-based implementation uses the top integer to track the index of the last element inserted. push increments top before insertion, and pop returns the element at top before decrementing it. Checks for overflow and underflow prevent boundary errors.

14. Demonstrate passing a simple object to a function by Value and by Reference â–¼ 
// Pass by Value vs. Pass by Reference (Objects)
#include <iostream>
using namespace std;

class Data {
public:
    int value;
    Data(int v) : value(v) {}
    
    void display(const char* msg) {
        cout << msg << ": Value = " << value << endl;
    }
};

// Pass by Value: creates a copy of the object
void modifyByValue(Data d) {
    d.value = 999; // Changes the COPY, not the original
    d.display("Inside modifyByValue");
}

// Pass by Reference: operates on the original object
void modifyByReference(Data& d) {
    d.value = 1000; // Changes the ORIGINAL object
    d.display("Inside modifyByReference");
}

int main() {
    Data myData(100);
    myData.display("Original");

    cout << "\n--- Testing Pass by Value ---" << endl;
    modifyByValue(myData);
    myData.display("After modifyByValue"); // Original remains 100

    cout << "\n--- Testing Pass by Reference ---" << endl;
    modifyByReference(myData);
    myData.display("After modifyByReference"); // Original is changed to 1000
    
    return 0;
}

Explanation:

Pass by Value creates a temporary copy of the object, so changes inside the function are lost when the function returns. Pass by Reference (using &) passes the memory address, allowing the function to directly access and modify the original object, making the changes permanent.

16. Merge two sorted integer arrays into a third sorted array â–¼ 
// Merge Two Sorted Arrays
#include <iostream>
using namespace std;

void mergeSortedArrays(const int arr1[], int size1, 
                       const int arr2[], int size2, 
                       int result[]) {
    int i = 0; // Pointer for arr1
    int j = 0; // Pointer for arr2
    int k = 0; // Pointer for result array

    // Merge elements while both arrays have elements
    while (i < size1 && j < size2) {
        if (arr1[i] < arr2[j]) {
            result[k++] = arr1[i++];
        } else {
            result[k++] = arr2[j++];
        }
    }

    // Copy remaining elements of arr1 (if any)
    while (i < size1) {
        result[k++] = arr1[i++];
    }

    // Copy remaining elements of arr2 (if any)
    while (j < size2) {
        result[k++] = arr2[j++];
    }
}

void displayArray(const int arr[], int size) {
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int arrA[] = {1, 3, 5, 7};
    int arrB[] = {2, 4, 6, 8, 10};
    int sizeA = 4;
    int sizeB = 5;
    const int TOTAL_SIZE = sizeA + sizeB;
    int mergedArr[TOTAL_SIZE]; // Needs to be large enough

    mergeSortedArrays(arrA, sizeA, arrB, sizeB, mergedArr);

    cout << "Merged Array: ";
    displayArray(mergedArr, TOTAL_SIZE);
    
    return 0;
}

Explanation:

This program uses the standard merge technique (part of Merge Sort). It utilizes three pointers (i, j, k) to simultaneously iterate through both source arrays (A and B) and the destination array (C). It compares elements from A and B and places the smaller one into C, advancing only the pointer of the array from which the element was taken. Finally, it copies any remaining elements.

int countDigits(int n) {
    int count = 0;
    while (n != 0) {
        n /= 10;
        count++;
    }

bool isArmstrong(int n) {
    int originalNum = n;
    int numDigits = countDigits(n);
    int sum = 0;
    int temp = n;

    while (temp > 0) {
        int digit = temp % 10;
        // Sum of (digit raised to the power of number of digits)
        sum += pow(digit, numDigits);

    }

}

An Armstrong number is one that is equal to the sum of its own digits raised to the power of the number of digits.
